 create table if not exists empresa(
 	id_empresa bigint unsigned auto_increment primary key,
  cnpj        varchar(14)   not null,
 	nome        varchar(255) not null,
 	imagem      varchar(255) default null,
 	estado      varchar(25)  not null,
 	bairro      varchar(25)  not null,
 	endereco    varchar(25)  not null,
 	tipo        varchar(100) not null,
 	categoria   SET ("fornecedor","loja","outro"),
 	descricao   text  default null,
 	observacao  text default null,
 	email       varchar(255) not null,
 	senha       varchar(255) not null
 );

//
 create table if not exists produto(
  id_produto bigint unsigned auto_increment primary key,
  id_empresa bigint unsigned not null,
  nome varchar(255) not null,
  imagem varchar(255) default null,
  descricao varchar(255) not null,
  valor DECIMAL(65,2),
  parcelamento tinyint unsigned default null,
  promocao varchar(255) default null,
  frete set ("1","0"),
  CONSTRAINT id_empresa FOREIGN KEY (id_empresa) REFERENCES empresa(id_empresa)
  ON DELETE CASCADE
  ON UPDATE CASCADE
 );
